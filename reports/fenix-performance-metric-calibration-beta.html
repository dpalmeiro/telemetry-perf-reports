      AND DATE(submission_timestamp) <= DATE('2024-06-11')
      AND normalized_channel = "beta"
      AND normalized_app_name = "Firefox"
      AND metrics.timing_distribution.perf_largest_contentful_paint is not null
      AND ARRAY_LENGTH(ping_info.experiments) > 0
      AND mozfun.map.get_key(ping_info.experiments, "fenix-performance-metric-calibration-beta").branch is not null
),
android_data as (
    SELECT 
        normalized_os as segment,
        mozfun.map.get_key(ping_info.experiments, "fenix-performance-metric-calibration-beta").branch as branch,
        CAST(key as INT64)/1000000 AS bucket,
        value as count
    FROM `mozdata.fenix.metrics` as f
      CROSS JOIN UNNEST(metrics.timing_distribution.perf_largest_contentful_paint.values)
    WHERE
      DATE(submission_timestamp) >= DATE('2024-05-14')
      AND DATE(submission_timestamp) <= DATE('2024-06-11')
      AND normalized_channel = "beta"
      AND metrics.timing_distribution.perf_largest_contentful_paint is not null
      AND ARRAY_LENGTH(ping_info.experiments) > 0
      AND mozfun.map.get_key(ping_info.experiments, "fenix-performance-metric-calibration-beta").branch is not null
)
SELECT
    segment,
    branch,
    bucket,
    SUM(count) as counts
FROM
    (
        SELECT * FROM desktop_data
        UNION ALL
        SELECT * FROM android_data
    ) s
GROUP BY
  segment, branch, bucket
ORDER BY
  segment, branch, bucket
